-- Sxtyny's Easy GodHacker666
-- Game ID: 131832665672192

local Players = game:GetService("Players")
local TeleportService = game:GetService("TeleportService")
local LocalPlayer = Players.LocalPlayer
local TARGET_GAME_ID = 131832665672192

-- Criar ScreenGui
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "SxtynysHack"
ScreenGui.ResetOnSpawn = false
ScreenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

-- Função para criar UI principal (quando está no jogo certo)
local function createMainUI()
    -- Frame principal
    local Frame = Instance.new("Frame")
    Frame.Size = UDim2.new(0, 400, 0, 180)
    Frame.Position = UDim2.new(0.5, -200, 0.5, -90)
    Frame.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    Frame.BorderSizePixel = 0
    Frame.Parent = ScreenGui

    -- Borda arredondada
    local UICorner = Instance.new("UICorner")
    UICorner.CornerRadius = UDim.new(0, 16)
    UICorner.Parent = Frame

    -- Borda azul
    local UIStroke = Instance.new("UIStroke")
    UIStroke.Color = Color3.fromRGB(0, 120, 255)
    UIStroke.Thickness = 3
    UIStroke.Parent = Frame

    -- Título
    local Title = Instance.new("TextLabel")
    Title.Size = UDim2.new(1, 0, 0, 45)
    Title.Position = UDim2.new(0, 0, 0, 10)
    Title.BackgroundTransparency = 1
    Title.Text = "Sxtyny's Easy GodHacker666"
    Title.TextColor3 = Color3.fromRGB(255, 255, 255)
    Title.TextSize = 20
    Title.Font = Enum.Font.GothamBold
    Title.Parent = Frame

    -- Descrição
    local Desc = Instance.new("TextLabel")
    Desc.Size = UDim2.new(1, -20, 0, 70)
    Desc.Position = UDim2.new(0, 10, 0, 55)
    Desc.BackgroundTransparency = 1
    Desc.Text = 'This script deletes the spawner and stuff that can kill you so you can play without dying and get the skin, when the text "Not Done" turns into "Done" you can join'
    Desc.TextColor3 = Color3.fromRGB(200, 200, 200)
    Desc.TextSize = 11
    Desc.Font = Enum.Font.Gotham
    Desc.TextWrapped = true
    Desc.Parent = Frame

    -- Status
    local Status = Instance.new("TextLabel")
    Status.Name = "Status"
    Status.Size = UDim2.new(1, 0, 0, 30)
    Status.Position = UDim2.new(0, 0, 0, 135)
    Status.BackgroundTransparency = 1
    Status.Text = "Not Done"
    Status.TextColor3 = Color3.fromRGB(255, 50, 50)
    Status.TextSize = 18
    Status.Font = Enum.Font.GothamBold
    Status.Parent = Frame

    return Frame, Status
end

-- Função para criar UI de joiner (quando NÃO está no jogo certo)
local function createJoinerUI()
    local Frame = Instance.new("Frame")
    Frame.Size = UDim2.new(1, 0, 1, 0)
    Frame.Position = UDim2.new(0, 0, 0, 0)
    Frame.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    Frame.BorderSizePixel = 0
    Frame.Parent = ScreenGui

    local Label = Instance.new("TextLabel")
    Label.Name = "JoinerLabel"
    Label.Size = UDim2.new(1, -40, 0, 80)
    Label.Position = UDim2.new(0, 20, 0.5, -40)
    Label.BackgroundTransparency = 1
    Label.Text = "JOINER | STATUS: JOINING GAME ID: " .. TARGET_GAME_ID
    Label.TextColor3 = Color3.fromRGB(255, 255, 255)
    Label.TextSize = 28
    Label.Font = Enum.Font.GothamBold
    Label.TextWrapped = true
    Label.Parent = Frame

    return Frame, Label
end

-- Verificar game ID atual
local currentGameId = game.GameId

if currentGameId == TARGET_GAME_ID then
    -- ✅ Está no jogo certo
    ScreenGui.Parent = LocalPlayer:WaitForChild("PlayerGui")
    local frame, statusLabel = createMainUI()

    -- Deletar spawners e cold
    local success = pcall(function()
        local playerGui = LocalPlayer:WaitForChild("PlayerGui")
        local mainGame = playerGui:WaitForChild("mainGame")
        local realGame = mainGame:WaitForChild("realGame")

        -- Deletar "spawners"
        local spawners = realGame:FindFirstChild("spawners")
        if spawners then
            spawners:Destroy()
        end

        -- Deletar "cold"
        local cold = realGame:FindFirstChild("cold")
        if cold then
            cold:Destroy()
        end
    end)

    -- Atualizar status
    if success then
        statusLabel.Text = "Done"
        statusLabel.TextColor3 = Color3.fromRGB(50, 255, 80)
    else
        statusLabel.Text = "Error - try rejoining"
        statusLabel.TextColor3 = Color3.fromRGB(255, 165, 0)
    end

else
    -- ❌ Não está no jogo certo — mostrar joiner
    ScreenGui.Parent = LocalPlayer:WaitForChild("PlayerGui")
    local frame, label = createJoinerUI()

    -- Tentar teleportar
    local teleportSuccess = pcall(function()
        TeleportService:Teleport(TARGET_GAME_ID, LocalPlayer)
    end)

    if not teleportSuccess then
        -- Falhou — atualizar texto e kickar
        label.Text = "JOINER | STATUS: JOINER NOT WORKING JOIN THE GAME MANUALLY"

        task.wait(5)

        LocalPlayer:Kick('Join "Hide in the attic from john bob and george hampleton" | By Sxtyny')
    end
end

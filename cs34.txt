-- Sxtyny's Easy GodHacker666

local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local TARGET_PLACE_ID = 131832665672192

if game.PlaceId ~= TARGET_PLACE_ID then
    LocalPlayer:Kick("join the right game")
    return
end

local old = LocalPlayer.PlayerGui:FindFirstChild("SxtynysHack")
if old then old:Destroy() end

local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "SxtynysHack"
ScreenGui.ResetOnSpawn = false
ScreenGui.DisplayOrder = 999999
ScreenGui.Parent = LocalPlayer.PlayerGui

local Frame = Instance.new("Frame")
Frame.Size = UDim2.new(0, 400, 0, 180)
Frame.Position = UDim2.new(0.5, -200, 0.5, -90)
Frame.BackgroundColor3 = Color3.new(0, 0, 0)
Frame.BorderSizePixel = 0
Frame.Parent = ScreenGui
Frame.Active = true

Instance.new("UICorner", Frame).CornerRadius = UDim.new(0, 16)

local stroke = Instance.new("UIStroke", Frame)
stroke.Color = Color3.fromRGB(0, 120, 255)
stroke.Thickness = 3

local title = Instance.new("TextLabel", Frame)
title.Size = UDim2.new(1, 0, 0, 45)
title.Position = UDim2.new(0, 0, 0, 10)
title.BackgroundTransparency = 1
title.Text = "Sxtyny's Easy GodHacker666"
title.TextColor3 = Color3.new(1, 1, 1)
title.TextSize = 20
title.Font = Enum.Font.GothamBold

local desc = Instance.new("TextLabel", Frame)
desc.Size = UDim2.new(1, -20, 0, 70)
desc.Position = UDim2.new(0, 10, 0, 55)
desc.BackgroundTransparency = 1
desc.Text = 'This script deletes the spawner and stuff that can kill you so you can play without dying and get the skin, when the text "Not Done" turns into "Done" you can join'
desc.TextColor3 = Color3.fromRGB(200, 200, 200)
desc.TextSize = 11
desc.Font = Enum.Font.Gotham
desc.TextWrapped = true

local status = Instance.new("TextLabel", Frame)
status.Size = UDim2.new(1, 0, 0, 30)
status.Position = UDim2.new(0, 0, 0, 135)
status.BackgroundTransparency = 1
status.Text = "Not Done"
status.TextColor3 = Color3.fromRGB(255, 50, 50)
status.TextSize = 18
status.Font = Enum.Font.GothamBold

-- DRAG
local dragging, dragStart, startPos
Frame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = Frame.Position
    end
end)

Frame.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = false
    end
end)

game:GetService("UserInputService").InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - dragStart
        Frame.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end)

task.wait(4)

local ok = pcall(function()
    local pg = LocalPlayer.PlayerGui
    local mg = pg:WaitForChild("maingame", 15)  -- ✅ minúsculo
    local rg = mg:WaitForChild("realgame", 15)  -- ✅ minúsculo

    if rg:FindFirstChild("spawners") then
        rg.spawners:Destroy()
    end
    if rg:FindFirstChild("cold") then
        rg.cold:Destroy()
    end
end)

if ok then
    status.Text = "Done"
    status.TextColor3 = Color3.fromRGB(50, 255, 80)
else
    status.Text = "Error"
    status.TextColor3 = Color3.fromRGB(255, 165, 0)
end
